168 function! StatuslineTrailingSpaceWarning()
169   if !exists("b:statusline_trailing_space_warning")
170     if search('\s\+$', 'nw') != 0
171       let b:statusline_trailing_space_warning = 'TWS'
172     else
173       let b:statusline_trailing_space_warning = ''
174     endif
175   endif
176   return b:statusline_trailing_space_warning
177 endfunction
178 
179 function! LocListCountSevere()
180   let ll_count = 0
181   for entry in getloclist(0)
182     if entry.type == "E" || entry.type == "W" || entry.type == ""
183       let ll_count = ll_count + 1
184     endif
185   endfor
186   if ll_count > 0
187     return "[LL: " . ll_count . "]"
188   else
189     return ""
190   endif
191 endfunction

243 function! Preserve(command)
244   let _s=@/
245   let l = line(".")
246   let c = col(".")
247   execute a:command
248   let @/=_s
249   call cursor(l, c)
250 endfunction
